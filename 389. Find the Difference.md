class Solution {
public:
    char findTheDifference(string s, string t) {
        unordered_map<char, int> m;
        for(char ch : s) {
            ++m[ch];
        }

        for(char ch : t) {
            if(0 == m[ch]) {
                return ch;
            }
            --m[ch];
        }

        return 'a';
    }
};
