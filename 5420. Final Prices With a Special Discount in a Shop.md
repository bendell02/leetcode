class Solution {
public:
    vector<int> finalPrices(vector<int>& prices) {
        int len = prices.size();
        if(len <= 0) { return {}; }
        
        vector<int> rlt(len);
        for(int i=0; i<len; i++) {
            int discount = 0;
            for(int j=i+1; j<len; j++) {
                if(prices[j]<=prices[i]) {
                    discount = prices[j];
                    break;
                }
            }
            rlt[i] = prices[i] - discount;
        }
        
        return rlt;
    }
};
